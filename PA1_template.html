                if (is.na(actUpdated[i,"steps"]))
                        {       stepName = as.character(actUpdated[i,"interval"])
                                mean1 =summaryStepsPerInterval[[stepName]]["Mean"]

                                max1 =summaryStepsPerInterval[[stepName]]["Max."]
                                stepValue = round(abs (rnorm(1)) * (max1-min1),0)
